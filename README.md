a = input('Digite algo:')
print("o tipo primitivo desse valor é:", type(a))
print('so tem espaço? ', a.isspace())
print("é um numero?", a.isnumeric())
print("é alfanumerico?", a.isalnum())
print(' esta em maiuscula', a.isupper())
print('esta em minuscula', a.islower())
